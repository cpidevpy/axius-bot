async function loadBots() { const res = await fetch('/api/accounts'); const accounts = await res.json(); const container = document.getElementById('botsList'); container.innerHTML = Object.entries(accounts).map(([id, acc]) => `
${acc.name}
${acc.token?.substring(0,20)}...
`).join(''); } async function addBot() { const name = document.getElementById('botName').value; const token = document.getElementById('botToken').value; if (!name || !token) return alert('Заполните поля'); const id = Date.now().toString(); await fetch('/api/accounts', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id, name, token }) }); document.getElementById('botName').value = ''; document.getElementById('botToken').value = ''; loadBots(); } async function deleteBot(id) { if (!confirm('Удалить бота?')) return; await fetch(`/api/accounts/${id}`, { method: 'DELETE' }); loadBots(); } loadBots();
